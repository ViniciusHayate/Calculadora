    private void BOTAO_RESULTADO_ActionPerformed(java.awt.event.ActionEvent evt) {                                                 

        valor2 = Double.parseDouble(txtResultado.getText());
        txtResultado.setText("");

        switch (sinal) {
            case "somar":
            resultado = valor1 + valor2;
            txtResultado.setText(String.valueOf(resultado));
            break;

            case "sub":
            resultado = valor1 - valor2;
            txtResultado.setText(String.valueOf(resultado));
            break;
            case "multiplicar":
            resultado = valor1 * valor2;
            txtResultado.setText(String.valueOf(resultado));
            break;
            case "dividir":
            resultado = valor1 / valor2;
            txtResultado.setText(String.valueOf(resultado));
        }
    }
